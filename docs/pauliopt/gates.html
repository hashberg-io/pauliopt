 11class Gate(ABC):
 12    """Base class for quantum gates."""
 13
 14    def __init__(self, *qubits):
 15        self.qubits = qubits
 16        self.name = self.__class__.__name__
 17        if len(qubits) != self.n_qubits:
 18            name, n_qubits = self.name, self.n_qubits
 19            raise ValueError(f"{name} takes {n_qubits} qubits, not {len(qubits)}.")
 20
 21    def __repr__(self) -> str:
 22        args = map(repr, self.qubits)
 23        return f"{self.name}({', '.join(args)})"
 24
 25    @property
 26    def width(self):
 27        """Width of gate used for drawing."""
 28        if getattr(self, "draw_as_zx", False):
 29            n_columns = max(self._gen_columns())
 30            return n_columns * 20 + 40
 31
 32        vscale = 1.0
 33        line_height = int(ceil(30 * vscale))
 34        min_qubit = min(self.qubits)
 35        max_qubit = max(self.qubits)
 36        text = repr(self)
 37        box_height = max(30, (max_qubit - min_qubit) * line_height + 10)
 38        box_width = max(8 * len(text), box_height * 1.5)
 39        return int(box_width)
 40
 41    def _gen_columns(self):
 42        n_columns = 0
 43        gate_cls = None
 44        # shift by 3 columns if a multi-legged gadget was drawn
 45        big_gadget_drawn = False
 46        for g in self.decomp:
 47            if gate_cls != (type(g), getattr(g, "basis", None)):
 48                gate_cls = (type(g), getattr(g, "basis", None))
 49                n_columns += 3 if big_gadget_drawn else 1
 50                big_gadget_drawn = False
 51            if isinstance(g, PhaseGadget) and len(g.qubits) > 1:
 52                big_gadget_drawn = True
 53            yield n_columns
 54        n_columns += 3 if big_gadget_drawn else 1
 55        big_gadget_drawn = False
 56        yield n_columns
 57
 58    def draw_zx(self, builder, base, row_width, params):
 59        n_columns = max(self._gen_columns())
 60        step = row_width / n_columns
 61
 62        busy = set()
 63        for g, column_idx in zip(self.decomp, self._gen_columns()):
 64            x, y = base[0] + column_idx * step, base[1]
 65
 66            if len(g.qubits) > 1:
 67                # gadget body blocks text
 68                busy.add(min(g.qubits) + 1)
 69            text_above = True
 70            if len(g.qubits) == 1 and tuple(g.qubits)[0] in busy:
 71                text_above = False
 72            if isinstance(g, PhaseGadget):
 73                draw_gadget(builder, (x, y), row_width, params, g, text_above)
 74            else:
 75                g.draw(builder, base, row_width, params)
 76
 77    def draw_box(self, builder, base, row_width, params):
 78        line_height = params["line_height"]
 79        font_size = params["font_size"]
 80
 81        min_qubit = min(self.qubits)
 82        max_qubit = max(self.qubits)
 83        x = base[0] + row_width / 2
 84        y = base[1] + ((min_qubit + max_qubit) / 2 + 1) * line_height
 85        text = repr(self)
 86        box_height = max(30, (max_qubit - min_qubit) * line_height + 10)
 87        box_width = max(8 * len(text), box_height * 1.5)
 88
 89        builder.rect((x, y), box_width, box_height, "#FFFFFF")
 90        builder.text((x, y), text, font_size=font_size, center=True)
 91
 92    def draw(self, builder, base, row_width, params):
 93        # builder.line((base[0], 0), (base[0], 100))
 94        # builder.line((base[0] + row_width, 0), (base[0] + row_width, 100))
 95        if getattr(self, "draw_as_zx", False):
 96            self.draw_zx(builder, base, row_width, params)
 97        else:
 98            self.draw_box(builder, base, row_width, params)
 99
100    @property
101    def gadgets(self):
102        """List of gadgets used to implement this gate."""
103        if not hasattr(self, "decomp"):
104            raise NotImplementedError
105
106        gadgets = [
107            g if isinstance(g, (ZHead, XHead)) else g.gadgets for g in self.decomp
108        ]
109        return gadgets

531def draw_gadget(builder, base, row_width, params, gadget, text_above=True):
532    line_height = params["line_height"]
533    text_off_x, text_off_y = params["text_off"]
534    font_size = params["font_size"]
535    r = params["r"]
536
537    if gadget.basis == "Z":
538        color1, color2 = params["zcolor"], params["xcolor"]
539    elif gadget.basis == "X":
540        color1, color2 = params["xcolor"], params["zcolor"]
541    else:
542        raise ValueError
543
544    step = row_width / 4
545    body = min(gadget.qubits) + 0.5
546    x = base[0]
547    x_body = x + step
548    y_body = base[1] + (body + 1) * line_height
549
550    if len(gadget.qubits) > 1:
551        for q in gadget.qubits:
552            y = base[1] + (q + 1) * line_height
553            builder.line((x, y), (x_body, y_body))
554            builder.circle((x, y), r, color1)
555        builder.line((x_body, y_body), (x_body + step, y_body))
556        builder.circle((x_body, y_body), r, color2)
557        builder.circle((x_body + step, y_body), r, color1)
558
559        text_x = x_body + step + text_off_x
560        text_y = y_body
561    else:
562        assert len(gadget.qubits) == 1
563        q = tuple(gadget.qubits)[0]
564        y = base[1] + (q + 1) * line_height
565        builder.circle((x, y), r, color1)
566
567        text_x = x + text_off_x
568        if text_above:
569            text_y = y + text_off_y
570        else:
571            text_y = y - text_off_y
572
573    builder.text((text_x, text_y), str(gadget.angle), font_size=font_size)
